app 柜员管理

bo 用户
end

bo 行员 extends 用户
  编号
  姓名
  性别
  入职状态 // [实习,在职,离职]
  工作状态 // [正常,休假]
  登录状态 // [在线,离线]
  // ...
  年龄 = 今天 - 生日
  角色 : 角色
  预留指纹 : 指纹
end

bp 行员::增加
  年龄 >= 18
end

bo 指纹
  手指 // [左手拇指,左手食指,...右手小指]
  指纹
  行员 : 行员
end

bo 角色
  编号
  分类
  名称
  菜单 : 菜单
  交易 : 交易
end

bo 菜单
  编号
  级别
  叶子
  名称
  子菜单 : 菜单
end

bo 个人客户 extends 用户
  编号
end

bp 行员登录
 exists 指纹
 exists 行员 // 这两句业务人员可能写不来
 any 行员.登录指纹 == 行员.预留指纹 // 执行起来是一个挨个比对的过程，要支持修饰才可
 行员.工作状态 == 正常
 行员.所属网段 == 登录网段
 行员.登录状态 = 在线
 登录日志.增加 // 这句可放在面向方面
end

bp 行员签退
 行员.登录状态 == 在线
 行员.登录状态 = 离线
 登录日志.增加
end

bp 交易权限检查
  any 交易 == 行员.交易
  // 金额，累计金额，累计次数 (配置?)
end
